/*

â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
Purpose: Generate a unique order ID
Returns: String (e.g., "#123456")
Usage: Called in showSuccessModal()
*/

function generateOrderId() {
    return '#' + Math.floor(Math.random() * 1000000).toString().padStart(6, '0');
}

/*
OUTPUT EXAMPLE:
#123456
#987654
#445231

/*

â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
Purpose: Convert payment method value to display name
Parameter: value (string) - payment method value
Returns: String (display name)
Usage: Called in showSuccessModal()

Supported Values:
- 'cod' â†’ 'Cash on Delivery'
- 'easypaisa' â†’ 'Easypaisa'
- 'jazzcash' â†’ 'JazzCash'

function getPaymentMethodName(value) {
    const methods = {
        'cod': 'Cash on Delivery',
        'easypaisa': 'Easypaisa',
        'jazzcash': 'JazzCash'
    };
    return methods[value] || 'Cash on Delivery';
}

/*
USAGE EXAMPLES:

/*
FUNCTION 3: validateCheckoutForm()
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
Purpose: Validate all required form fields
Returns: Boolean (true if valid, false if invalid)
Checks:
  âœ“ Full Name (not empty)
  âœ“ Email (valid format with @)
  âœ“ Phone (at least 10 digits)
  âœ“ Address (not empty)
  âœ“ City (not empty)
  âœ“ Postal Code (not empty)

If any field is invalid:
  - Shows alert message
  - Returns false
  - Form submission prevented

Usage: Called in "Confirm Order" button click handler
*/

function validateCheckoutForm() {
    const fullName = document.getElementById('fullName')?.value.trim();
    const email = document.getElementById('email')?.value.trim();
    const phone = document.getElementById('phone')?.value.trim();
    const address = document.getElementById('address')?.value.trim();
    const city = document.getElementById('city')?.value.trim();
    const zipcode = document.getElementById('zipcode')?.value.trim();

    if (!fullName) {
        alert('Please enter your full name');
        return false;
    }
    if (!email || !email.includes('@')) {
        alert('Please enter a valid email address');
        return false;
    }
    if (!phone || phone.length < 10) {
        alert('Please enter a valid phone number');
        return false;
    }
    if (!address) {
        alert('Please enter your delivery address');
        return false;
    }
    if (!city) {
        alert('Please enter your city');
        return false;
    }
    if (!zipcode) {
        alert('Please enter your postal code');
        return false;
    }
    return true;
}

VALIDATION EXAMPLES:
Input: Form with empty name
Output: alert('Please enter your full name') â†’ return false

Input: Email "invalid.com" (no @)
Output: alert('Please enter a valid email address') â†’ return false

Input: Phone "123"
Output: alert('Please enter a valid phone number') â†’ return false

Input: All fields filled correctly
Output: return true
*/

function showSuccessModal(paymentMethod) {
    const modal = document.getElementById('successModal');
    const orderId = generateOrderId();
    const totalAmount = 'Rs. 2,586';

    // Update modal content
    document.getElementById('orderId').textContent = orderId;
    document.getElementById('totalAmount').textContent = totalAmount;
    document.getElementById('paymentMethod').textContent = getPaymentMethodName(paymentMethod);

    // Show modal
    modal.classList.add('show');
}

/*

â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
Purpose: Initialize all checkout page event listeners
Called: On page load (DOMContentLoaded)

Sets Up:
1. "Confirm Order" button click handler
   - Validates form
   - Gets selected payment method
   - Shows success modal if valid

2. Modal close button (X) click handler
   - Calls closeSuccessModal()

3. Modal background click handler
   - Closes modal if clicked outside

Usage: Automatically called when checkout.html loads
*/

function initCheckout() {
    const confirmOrderBtn = document.getElementById('confirmOrderBtn');
    const checkoutForm = document.getElementById('checkoutForm');
    const successModal = document.getElementById('successModal');
    const modalClose = document.querySelector('.modal-close');

    // Handle confirm order button
    if (confirmOrderBtn) {
        confirmOrderBtn.addEventListener('click', function(e) {
            e.preventDefault();

            // Validate form
            if (!validateCheckoutForm()) {
                return;
            }

            // Get selected payment method
            const paymentMethod = document.querySelector('input[name="paymentMethod"]:checked')?.value || 'cod';

            // Show success modal
            showSuccessModal(paymentMethod);
        });
    }

    // Close modal when X is clicked
    if (modalClose) {
        modalClose.addEventListener('click', closeSuccessModal);
    }

    // Close modal when clicking outside
    if (successModal) {
        successModal.addEventListener('click', function(e) {
            if (e.target === successModal) {
                closeSuccessModal();
            }
        });
    }
}
